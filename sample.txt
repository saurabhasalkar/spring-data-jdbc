public Page<WorkflowInstanceDto> getRunningWorkflows(
        Pageable pageable,
        String workflowId,
        String businessDate,
        String initiatedBy
) {
    LocalDate businessDateFilter = null;

    if (businessDate != null && !businessDate.trim().isEmpty()) {
        try {
            businessDateFilter = LocalDate.parse(businessDate, DateTimeFormatter.ISO_LOCAL_DATE);
        } catch (DateTimeParseException ex) {
            throw new IllegalArgumentException("businessDate must be yyyy-MM-dd. Received: " + businessDate);
        }
    }

    String wfId = (workflowId == null || workflowId.trim().isEmpty()) ? null : workflowId.trim();
    String initiatedByFilter = (initiatedBy == null || initiatedBy.trim().isEmpty()) ? null : initiatedBy.trim();

    return workflowExecutionRepository.findRunningWorkflows(pageable, wfId, businessDateFilter, initiatedByFilter);
}
