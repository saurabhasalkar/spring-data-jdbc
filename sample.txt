@Test
void calculateNthWorkingDay_throwsWhenCalendarNotFound() {
    // Arrange
    String missingId = "MISSING";
    when(calendarRepository.findById(missingId)).thenReturn(Optional.empty());

    // Act + Assert  âœ… NOTE: pass the exception class as the first arg
    IllegalArgumentException ex = assertThrows(
            IllegalArgumentException.class,
            () -> realService.calculateNthWorkingDay(missingId, LocalDate.of(2025, 8, 1), 3)
    );

    assertTrue(ex.getMessage().toLowerCase().contains("calendar not found"));
}
