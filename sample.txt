package com.example.workdays.dto;

import java.time.LocalDate;

public class WorkingDayResponse {
    private int year;
    private int month; // 1-12
    private int nth;
    private LocalDate date;
    private String isoDate; // convenience

    public WorkingDayResponse(int year, int month, int nth, LocalDate date) {
        this.year = year;
        this.month = month;
        this.nth = nth;
        this.date = date;
        this.isoDate = date.toString();
    }

    public int getYear() { return year; }
    public int getMonth() { return month; }
    public int getNth() { return nth; }
    public LocalDate getDate() { return date; }
    public String getIsoDate() { return isoDate; }
}
